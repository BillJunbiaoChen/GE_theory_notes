\section{Arrow-Debreu Equilibrium (with Production)}
\noindent \textbf{Definition.} \\
An Arrow-Debreu Equilibrium is $(x, y, \phi) 
\in \mathbb{R}^{L(S+1)I}_{+} \times \Upsilon \times \mathbb{R}^{L(S+1)}_{++}$
(N.B.: $y = [y_h], \text{ with } y_h = (-k^h, f^h(k^h))$.) such that 
\begin{enumerate}
    \item Given $\phi$, each individual solves the utility maximization pb:
    \begin{align*}
        \max_{x^i_0, x^i_s} \quad & u^i(x^i_0) + \sum_s \text{Prob}_s u^i(x^i_s) \\
        \text{subject to: } \quad & \phi_0 (x^i_0 - w^i_0) + \sum_s \phi_s (x^i_s - w^i_s)  + \sum_h \delta^{ih} k_h = 0 
    \end{align*}
    \item Each firm solves the following pb:
    \begin{align*}
        \max_{k^h} \quad & - \phi_0 k^h + \textcolor{blue}{\sum_s \phi_s} f^h(k_h;s)
    \end{align*}
    \item Market clears%
    \footnote{Since $y^h = [-k^h, f^h(k^h)]$, this condition is the same as 
    \begin{align*}
        \sum_i x_0^i & \leq \sum_i w_0^i - \sum_h k^h \\
        \sum_i x_s^i & \leq \sum_i w_s^i + \sum_h f^h(k^h;s)
    \end{align*}
    }
    \begin{align*}
        \sum_i x^i \leq \sum_i w^i + \sum_h y^h
    \end{align*}
    where $x^i \in \mathbb{R}^{L(S+1)}_{+}$.
    \item Profits are 
    \[
    \pi^h = - \phi_0 k^h + \sum_s \phi_s f^h(k^h;s)
    \]
\end{enumerate}

\begin{tcolorbox}
    \begin{proposition}
        Let $(\theta, k)$ be an equilibrium allocation. 
        Consider an enlarged economy where agents can hold equity 
        $\theta_e^i$ in the original firm $f(k)$ 
        and $\gamma_e^i$ in an off-equilibrium firm $f(k')$.
        Then:
        \begin{itemize}
            \item Under \textbf{Rational Conjectures}, $\gamma^i_e = 0$ for all $i$.
            \item Under \textbf{DrÃ¨ze Conjectures}, the condition $\gamma^i_e = 0$ for all $i$ is not guaranteed to hold (trade may occur).
        \end{itemize}
    \end{proposition}
\end{tcolorbox}
\noindent \textbf{Proof (Sketch):}

Mathematically, this result is because

\begin{align*}
\mathbb{E}[m^i p \circ f^h({k^h}')] & \leq Q^h({k^h}'), \quad \forall i \\
\exists j, \text{ such that } \quad \mathbb{E}[m^j p \circ f^h({k^h}')] & > D^h({k^h}') 
\end{align*}

In words, the proposition says:
\begin{itemize}
    \item In the case where the equity is priced using the highest-bidder's evaluation,
    no one is going to buy the off-equilibrium firms if the highest-bidder doesn't buy it.
    \item However, if the equity of the off-equilibrium firms are priced using equilibrium holders' values,
    there might exist a potential buyer who values these firms more and thus want to buy them.
\end{itemize}

\begin{tcolorbox}
    \begin{proposition}
        Assuming $J = 0, L = H = 1$, competitive equilibria of the FME with production are constrained P.E.
    \end{proposition}
\end{tcolorbox}
The key of the proof relies on the optimization of the firm.
\\
\noindent \textbf{Proof (Sketch):}
Let $(x, \theta, k)$ be a competitive equilibrium. 
Suppose there exists $({x^i_0}', {\theta^i}') \succeq_{P} ({x^i_0}, {\theta^i})$, then by 
the Law of Revealed Preference, we have 
\[
{x^i_0}' + Q(k') {\theta^i}' \geq {x^i_0} + Q(k) {\theta^i} \forall i
\]
where some $i$ the inequality is strict. 
Now using the budget constraint and the consistency in the eqlm,  
($x^i_0 = w^i_0 + ( -k + Q(k))\theta^i_0 - Q(k) \theta^i )$, we have 
\begin{align*}
{x^i_0}' + Q(k') {\theta^i}' & \geq {x^i_0} + Q(k) {\theta^i}  \\ 
& =  w^i_0 + ( -k + Q(k))\theta^i_0 - Q(k) \theta^i + Q(k) {\theta^i}  \\ 
& =  w^i_0 + ( -k + Q(k))\theta^i_0  
\end{align*}
Summing over $i$, and applying the market clearing condition ($\sum_i {\theta^i}' = 1$), then we have 
the following strict inequality
\[
Q(k') + \sum_i {x^i_0}' >  -k + Q(k) + \sum_i w^i_0 
\]
Now here comes the critical step:
Due to the optimization of the firm, we have
\[
-k + Q(k) \geq -k' + Q(k')
\]
It follows that 
\[
Q(k') + \sum_i {x^i_0}' >  -k' + Q(k') + \sum_i w^i_0 
\]
Hence, we derive a contradiction:
$\sum_i {x^i_0}' + k' > \sum_i w^i_0$. \(\blacksquare\)

\section{Rational price conjectures}


\begin{tcolorbox}
\begin{proposition}
Define $Q^h(k^h) := \max_i \mathbb{E}[m^i p \circ f^h(k^h)]$, and 
$D^h(k^h) := \mathbb{E}[\sum_i \theta^i_h m^i p \circ f^h(k^h)]$, we have
\[
\text{FME with Rational Conjectures } (Q^h) \quad \overset{\Rightarrow}{\nLeftarrow} \quad \text{FME with Dreze Conjectures} (D^h)
\]
\end{proposition}
\noindent \textbf{Proof (Sketch):} Note that in equilibrium, 
these equity prices coincide with RC and Dreze Conjecture, namely, $Q^h({k^h}^*) = D^h({k^h}^*)$,
where ${k^h}^*$ is the equilibrium input choice of firm $h$.
However, away from equilibrium, we have 
\[
Q^h(k^h) \geq D^h(k^h)
\]
Hence, there are no profitable deviations from equilibrium under RC, (that is the $\Rightarrow$ part),
but there could be profitable deviations under Dreze (that is the $\nLeftarrow$ part).
\(\blacksquare\)    
\end{tcolorbox}


\section{Corporate Finance}
\subsection{FME with bonds and equity}
\noindent \textbf{Definition.} 
A competitive equilibrium of a Financial Market Economy with production, rational price conjecture, 
under 2 periods (assuming $I > 1$ individuals, 
$S$ states,  
$J = 1$ assets, and $H =1$ firm) is 
defined by 
prices and quantities 
$(x, z, \theta, B, k, q, Q, P) \in 
\mathbb{R}^{L(S+1)I}_{+} \times \mathbb{R}^{JI}_{+} \times \mathbb{R}^{HI}_{+} \times \mathbb{R}^{HI}_{+} 
\times \mathbb{R}^{LH}_{+} \times \mathbb{R}^{J}_{++} \times \mathbb{R}^{H}_{++} \times \mathbb{R}^{H}_{++}$, such that 
\begin{enumerate}
    \item Each consumer solves 
    \begin{align*}
        \max_{x^i, z^i, \theta^i, B^i} \quad & u^i(x^i_0) + \sum_{s \in S} \text{prob}_s u^i(x^i_s) \\
        \text{ subject to: } \quad &  x^i_0 - \omega^i_0 =  \left(-k + Q \right) \theta^i_0 - Q \theta^i - P B^i - q z^i \\
        & x^i_s - \omega^i_s = a_s z^i + \sum_h \theta^i_h \max\left\{0, f^h(k^h;s) - B_h \right\} + \sum_h B^i_h \min\left\{B_h, f^h(k^h;s)\right\} , \quad \forall s \in S
    \end{align*}
    \item Each firm solves 
    \begin{align*}
        \max_{k^h, B^h} \quad & -k^h + Q(k^h, B^h) + P(k^h, B^h) B^h
    \end{align*}
    \item No short sales:
    \begin{align*}
        \text{Goods: } \quad \sum_i (x^i - w^i) & = 0 \\
        \text{Assets: } \quad \sum_i z^i & = 0 \\
        \text{Equity: } \quad \sum_i \theta^i & = 1 
    \end{align*}
    \item Markets clear 
    \begin{align*}
        \text{Goods: } \quad \sum_i (x^i - w^i) & = 0 \\
        \text{Assets: } \quad \sum_i z^i & = 0 \\
        \text{Equity: } \quad \sum_i \theta^i & = 1 
    \end{align*}
    \item The price of equity satisfies rational conjectures (for in- and off- equilibrium paths)
    \begin{align*}
    Q^h (k^h, B^h) & = \max_i \mathbb{E}\bigg[m^i [\max\{0, f(k;s) - B \}] \bigg], \quad \forall h \\
    P^h (k^h, B^h) & = \max_i \mathbb{E}\bigg[m^i [\min \{1, \frac{f(k;s)}{B} \}] \bigg], \quad \forall h
    \end{align*}
    \item Consistency: $Q^h = Q^h(k^h, B^h)$ and  $P^h = P^h(k^h, B^h)$.
\end{enumerate}
