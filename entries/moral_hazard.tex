\section{Prescott-Townsend Competitive Equilibrium}
\noindent \textbf{Definition.} 
A competitive equilibrium in a Moral Hazard Economy is defined by 
an allocation $(x, e) \in \mathbb{R}^S \times E$, 
and prices $q \in \mathbb{R}^{SE}$ such that 
\begin{enumerate}
    \item Each consumer $i$ solves 
    \begin{align*}
        \max_{x^i, e} \quad & \sum_s \text{Prob}_s(e) u(x_s) - v(e) \\ 
        \text{subject to: } \quad & \sum_s q_s(e) (x_s - w_s ) = 0 
    \end{align*}
    \item Market clears 
    \[
    \sum_s \text{Prob}_s(e) (x^i_s - w^i_s) = 0,
    \]
    where due to the Law of Large Number, $\text{Prob}_s(e)$ is the fraction of 
    consumers choose effort $e$ when state $s$ is realized.
    \item Self-imposing incentive constraint:
    \[
    e \in \text{argmax}_\epsilon \sum_s \text{Prob}_s(\epsilon) u(x_s) - v(\epsilon), \quad \text{given} x
    \]
    \item Rational conjecture
    \[
        \text{Prob}_s(e) = q_s(e)
    \]
\end{enumerate}



\section{Radner Rational Expectation Equilibrium}
For an economy with $t = 0, 1$, $S$ states, $J$ assets, $L=1$ commodity, 
$I > 1$ agents with strictly positive endowments and 
strictly monotonic, vNM, and strictly quasiconcave preferences, and
$\Sigma^I$ signals.
A Radner Rational Expectation Equilibrium consists of 
$(x, z, q_\sigma, \phi) \in 
\mathbb{R}_{+}^{(S+1)I} \times \mathbb{R}_{+}^{JI} \times 
\mathbb{R}_{++}^{J |\Sigma|^I} \times \Phi$ 
such that 
\begin{enumerate}
    \item Given $w_i, A, q_\sigma, \phi, \sigma_i$, agent $i$ solves the following problem 
        \begin{align*}
            \max_{x_i, q_i} \ & U_i(x^i_0) + \sum_s \text{prob}_s^{\sigma_i, \phi^{-1}(q_\sigma)} U_i(x^i_s) \\ 
            \text{ subject to: }  & x^i_0 - w^i_0 + q_\sigma z_i = 0 \\ 
            &  x^i_s - w^i_s = a_s z_i, \quad \forall s 
        \end{align*}
    where $\text{prob}_s^{\sigma_i, \phi^{-1}(q_\sigma)}$ is the posterior beliefs about the state $s$ given 
    $i$'s signal $\sigma_i$ and her information about others' signals given the inverse price-map and the price $\phi^{-1}(q_\sigma)$.
    \item Markets clear 
        \begin{itemize}
            \item Assets (signal-dependent): $\sum_i z_i(q_\sigma, \phi) = 0, \forall \sigma \in \Sigma^I$
            \item Goods at $t = 0$: $\sum_i x^i_0 - w^i_0 = 0$
            \item Goods at $t = 1$: $\sum_i x^i_s - w^i_s = 0, \forall s$
        \end{itemize}
    \item Assets prices are consistent: $q_\sigma = \phi(\sigma)$.
\end{enumerate}

\subsection{Fully Revealing Equilibria}

\begin{definition}
A REE price map $\phi: \Sigma^I \to \mathbb{R}^J_+$ is \emph{fully revealing} if
\[
\phi(\sigma) \neq \phi(\sigma'), \quad \text{for any } \sigma, \sigma' \in \Sigma^I \text{ with } \sigma \neq \sigma'.
\]
Equivalently, $\phi$ is invertible: by observing $q_\sigma$, every agent can back out the joint signal profile $\sigma \in \Sigma^I$, so the posterior conditional on $(\sigma^i, \phi^{-1}(q_\sigma))$ collapses to the full-information posterior conditional on $\sigma$,
\[
\text{prob}^{\sigma^i, \phi^{-1}(q_\sigma)}_s = \text{prob}^{\sigma}_s.
\]
\end{definition}

\begin{theorem}[Radner 1979]
For a generic subset of economies parametrized by endowments $\omega \in \mathbb{R}^{(S+1)I}_{++}$, there exists a REE price map $\phi: \Sigma^I \to \mathbb{R}^J_+$ that is fully revealing.
\end{theorem}

\paragraph{Proof sketch (counting equations and unknowns).}
Because a fully revealing $\phi$ is invertible, every agent's posterior is the full-information posterior $\text{prob}^\sigma_s$. Consider the auxiliary economy in which agent $i$'s problem is
\begin{align*}
    \max_{(x^i, z^i) \in \mathbb{R}^{S+1}_+ \times \mathbb{R}^J} \quad & u^i(x_0^i) + \sum_{s \in S} \text{prob}^\sigma_s \, u^i(x_s^i) \\
    \text{subject to: } \quad & x_0^i + q_\sigma z^i \leq \omega_0^i, \\
                              & x_s^i \leq a_s z^i + \omega_s^i, \quad \forall s.
\end{align*}
Let $z^i(q_\sigma, \phi)$ denote the optimal portfolio and
$Z(q_\sigma, \phi) = \sum_{i \in I} z^i(q_\sigma, \phi)$
the aggregate excess demand for the $J$ assets.

We argue by contrapositive: suppose $\phi$ is \emph{not} fully revealing, i.e., there exist $\sigma \neq \sigma' \in \Sigma^I$ such that $\phi(\sigma) = \phi(\sigma')$. Then the following three conditions must hold simultaneously:
\begin{align*}
    Z(\phi(\sigma),  \phi)   &= 0, \\
    Z(\phi(\sigma'), \phi) &= 0, \\
    \phi(\sigma) &= \phi(\sigma').
\end{align*}
\emph{Count:} treating $\phi(\sigma), \phi(\sigma') \in \mathbb{R}^J$ as the unknowns, this is a system of $3J$ equations in $2J$ unknowns. By the Transversality Theorem applied to the endowment parameter $\omega \in \mathbb{R}^{(S+1)I}_{++}$, an over-determined system of this form generically (off a closed, measure-zero subset of $\omega$) has no solution. Hence generically no two distinct signal profiles deliver the same equilibrium price, and the REE is fully revealing. \hfill $\square$

\paragraph{Remarks.}
\begin{itemize}
    \item At a fully revealing REE, all agents share the same posterior $\text{prob}^\sigma_s$, so the equilibrium allocation is exactly the one in a full-information financial market economy: \emph{agents do not trade on account of information}, only on account of risk-sharing.
    \item The genericity logic is the same as the one used in incomplete markets (see Geanakoplos-Polemarchakis): a transversality count $3J$ vs.\ $2J$ pins down the result, even though we have not made the perturbation argument explicit.
    \item The result \emph{fails} when the signal space is too rich relative to the asset span. If $\sigma \in \mathbb{R}^{\Sigma I}$ is continuous and $\Sigma I > J$, the price map $\phi: \mathbb{R}^{\Sigma I} \to \mathbb{R}^J$ generically cannot be invertible by a dimensionality argument: $\phi$ maps a $\Sigma I$-dimensional set into a $J$-dimensional one (Theorem 6.2 / Allen 1981).
\end{itemize}

\subsection{Problem 6.1}
\paragraph{Problem.} Characterize Incentive constrained optima and competitive equilibria (with the consistency condition on price maps) of the moral hazard economy specialized so that
\[
S = \{1, 2\}; \quad e \in \{h, \ell\}; \quad v(h) > v(\ell); \quad \pi_1(h) > \pi_1(\ell); \quad \omega_1 > \omega_2.
\]
Set-up and characterize equilibria of this economy under the constraint that prices are linear (independent of $x$), $q_s(e(x)) = q_s$. Are equilibrium allocations in this case Incentive constraint efficient?

\paragraph{Solution.}
An IC optimum $(x^*, e^*)$ solves:
\[
\max_{x, e} \ \pi_1(e) u(x_1) + (1 - \pi_1(e)) u(x_2) - v(e)
\]
\[
\text{s.t.} \quad
\begin{cases}
\pi_1(e) (x_1 - \omega_1) + (1 - \pi_1(e)) (x_2 - \omega_2) = 0 \\
e \in \arg\max_\varepsilon \pi_1(\varepsilon) u(x_1) + (1 - \pi_1(\varepsilon)) u(x_2) - v(\varepsilon)
\end{cases}
\]
A competitive equilibrium $(x^*, e^*, q)$ clears markets and solves:
\[
\max_{x, e} \ \pi_1(e) u(x_1) + (1 - \pi_1(e)) u(x_2) - v(e)
\]
\[
\text{s.t.} \quad
\begin{cases}
q_1 (x_1 - \omega_1) + q_2 (x_2 - \omega_2) = 0 \\
e \in \arg\max_\varepsilon \pi_1(\varepsilon) u(x_1) + (1 - \pi_1(\varepsilon)) u(x_2) - v(\varepsilon)
\end{cases}
\]
Moreover, $q_s = \pi_s(e^*)$. Equilibria are in general not efficient. High effort might be supported in the IC optimum if $\pi_1(h)$ is sufficiently larger than $\pi_1(\ell)$ and $\omega_1$ is sufficiently larger than $\omega_2$. However, high effort cannot be supported in equilibrium because prices do not change with the choice of $x$.

\subsection{Problem 6.2}
\paragraph{Problem.} Write the moral hazard economy with idiosyncratic uncertainty, as in this section, but allowing for consumption at time $t = 0$. Be careful in writing the consistency condition on prices and compare it with Rational conjectures as we formulated them in the production economies with Incomplete markets.

\paragraph{Solution.}
A competitive equilibrium $(x^*, z^*, e^*, q)$ clears markets and solves:
\[
\max_{x, e, z} \ u(x_0) + \sum_{s \in S} \pi_s(e) u(x_s)
\]
\[
\text{s.t.} \quad
\begin{cases}
x_0 - \omega_0 + q_0(x) z \leq 0 \\
q_s(x) (x_s - \omega_s) \leq a_s z \\
e \in e(x) = \arg\max_\varepsilon u(x_0) + \sum_{s \in S} \pi_s(\varepsilon) u(x_s)
\end{cases}
\]
Moreover, $q_s(x) = \pi_s(e(x))$ and $q_0(x) = \sum_{s \in S} \pi_s(e(x)) m_s(x) a_s$ for all $x$.
